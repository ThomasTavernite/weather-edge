// ─── Store loaded data for modal reuse ───────────────────
let ALL_DATA = [];

function formatPrice(p) {
  if (p === null || p === undefined) return '--';
  const n = parseFloat(p);
  if (isNaN(n)) return '--';
  if (n < 1) return Math.round(n * 100) + '¢';
  return '$' + n.toFixed(2);
}

// ─── Load Overview ───────────────────────────────────────
async function loadOverview() {
  try {
    const res = await fetch('/api/overview');
    const json = await res.json();
    ALL_DATA = json.cities || [];

    // Update status bar
    document.getElementById('statusDot').className = 'status-dot live';
    document.getElementById('statusText').textContent = 'Live';
    if (json.updatedAt) {
      document.getElementById('updateTime').textContent = 'Updated ' + new Date(json.updatedAt).toLocaleTimeString();
    }

    // Split into categories
    const divergences = ALL_DATA.filter(c => c.signal?.type === 'DIVERGENCE');
    const aligned = ALL_DATA.filter(c => c.signal?.type === 'ALIGNED');
    const noData = ALL_DATA.filter(c => !c.signal);

    let html = '';

    if (divergences.length > 0) {
      html += `<div class="section-label">⚡ Potential Opportunities (${divergences.length})</div>`;
      html += '<div class="city-grid">' + divergences.map(renderCard).join('') + '</div>';
    }
    if (aligned.length > 0) {
      html += `<div class="section-label">✓ Market Aligned with Forecasts (${aligned.length})</div>`;
      html += '<div class="city-grid">' + aligned.map(renderCard).join('') + '</div>';
    }
    if (noData.length > 0) {
      html += `<div class="section-label">⏳ Awaiting Data (${noData.length})</div>`;
      html += '<div class="city-grid">' + noData.map(renderCard).join('') + '</div>';
    }

    if (!html) html = '<div class="loading-container"><h2>No active markets right now</h2><p>Kalshi weather markets typically open daily around 5 AM ET</p></div>';

    document.getElementById('mainContent').innerHTML = html;

  } catch (e) {
    console.error('Load failed:', e);
    document.getElementById('mainContent').innerHTML = `
      <div class="loading-container">
        <h2>Loading data...</h2>
        <p>Function is warming up. Retrying in 10 seconds...</p>
      </div>`;
    setTimeout(loadOverview, 10000);
  }
}

function renderCard(c) {
  const temp = c.currentTemp !== null ? Math.round(c.currentTemp) + '<span class="unit">°F</span>' : '--';
  const consensus = c.consensus?.high !== null ? c.consensus.high + '°F' : '--';

  let badge = '<span class="signal-badge nodata">NO DATA YET</span>';
  let detail = '';

  if (c.signal) {
    if (c.signal.type === 'DIVERGENCE') {
      badge = '<span class="signal-badge divergence">⚡ DIVERGENCE</span>';
      detail = `Forecasts → <strong>${c.signal.consensusBracket}</strong> (${formatPrice(c.signal.consensusBracketPrice)}) · Market favors <strong>${c.signal.marketFavorite}</strong> (${formatPrice(c.signal.marketFavoritePrice)})`;
    } else {
      badge = '<span class="signal-badge aligned">✓ ALIGNED</span>';
      detail = `Forecasts & market agree: <strong>${c.signal.marketFavorite}</strong> at ${formatPrice(c.signal.marketFavoritePrice)}`;
    }
  }

  return `
    <div class="city-card" onclick="openCity('${c.id}')">
      <div class="card-top">
        <div>
          <div class="city-name">${c.name}</div>
          ${badge}
        </div>
        <div style="text-align:right">
          <div class="current-temp">${temp}</div>
          <div class="current-desc">${c.currentDesc || 'Current'}</div>
        </div>
      </div>
      <div class="forecast-row">
        ${(c.forecasts || []).map(f => `
          <div class="forecast-chip">
            <span class="src">${f.source}</span>
            <span class="val">H: ${f.high != null ? f.high + '°' : '--'}</span>
          </div>
        `).join('')}
        ${c.consensus?.high != null ? `
          <div class="forecast-chip consensus">
            <span class="src">Consensus</span>
            <span class="val">${c.consensus.high}°F</span>
          </div>
        ` : ''}
      </div>
      ${detail ? `<div class="signal-detail">${detail}</div>` : ''}
      <a href="https://kalshi.com/category/climate" target="_blank" class="kalshi-link" onclick="event.stopPropagation()">Trade on Kalshi →</a>
    </div>`;
}

// ─── Modal (uses already-loaded data) ────────────────────
function openCity(id) {
  const c = ALL_DATA.find(d => d.id === id);
  if (!c) return;

  document.getElementById('modal').classList.add('active');
  document.getElementById('modalTitle').textContent = c.name;

  let html = '';

  // Current temp
  if (c.currentTemp !== null) {
    html += `<div class="consensus-row">
      <span class="consensus-label">Current Temp — ${c.currentDesc || 'NWS Station'}</span>
      <span class="consensus-val">${c.currentTemp}°F</span>
    </div>`;
  }

  // Forecasts
  html += '<div class="section-label" style="margin-top:0">Forecasts for Today</div>';
  html += '<div class="forecast-row" style="margin-bottom:20px">';
  (c.forecasts || []).forEach(f => {
    html += `<div class="forecast-chip">
      <span class="src">${f.source}</span>
      <span class="val">H: ${f.high != null ? f.high + '°' : '--'}  L: ${f.low != null ? f.low + '°' : '--'}</span>
    </div>`;
  });
  if (c.consensus?.high != null) {
    html += `<div class="forecast-chip consensus">
      <span class="src">Consensus</span>
      <span class="val">H: ${c.consensus.high}°  L: ${c.consensus.low || '--'}°</span>
    </div>`;
  }
  html += '</div>';

  // Kalshi brackets table
  if (c.kalshi?.markets?.length) {
    html += `<div class="section-label">Kalshi Market — ${c.kalshi.title || c.kalshi.subtitle || 'High Temp'}</div>`;
    html += '<table class="bracket-table"><thead><tr><th>Bracket</th><th>Price</th><th>Volume</th></tr></thead><tbody>';

    for (const m of c.kalshi.markets) {
      const price = parseFloat(m.lastPrice) || parseFloat(m.yesAsk) || 0;
      const isMatch = c.signal?.consensusBracket === m.subtitle;
      html += `<tr class="${isMatch ? 'consensus-match' : ''}">
        <td>${m.subtitle}${isMatch ? ' ← Forecast' : ''}</td>
        <td class="price-cell">${formatPrice(price)}</td>
        <td style="color:#64748b">${parseInt(m.volume || 0).toLocaleString()}</td>
      </tr>`;
    }
    html += '</tbody></table>';
    html += `<a href="https://kalshi.com/category/climate" target="_blank" class="kalshi-link">Trade This Market on Kalshi →</a>`;
  } else {
    html += '<p style="color:#64748b;margin-top:16px">No active Kalshi markets found for today.</p>';
  }

  document.getElementById('modalBody').innerHTML = html;
}

function closeModal() { document.getElementById('modal').classList.remove('active'); }

document.getElementById('modal').addEventListener('click', e => {
  if (e.target === document.getElementById('modal')) closeModal();
});
document.addEventListener('keydown', e => { if (e.key === 'Escape') closeModal(); });

// ─── Init ────────────────────────────────────────────────
loadOverview();
setInterval(loadOverview, 5 * 60 * 1000);
</script>
